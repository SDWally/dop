# 链表常见

## 链表去重

```
class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        unique_set = set([])
        root = ListNode(0, head)
        pre = root
        while head:
            val = head.val
            # print(val) 
            if val not in unique_set:
                unique_set.add(val)
                pre.next = head
                pre = pre.next
            else:
                pre.next = pre.next.next
            head = head.next
        return root.next
```

## 去除指定元素

```
class Solution:
    def removeElements(self, head: Optional[ListNode], val: int) -> Optional[ListNode]:
        root = ListNode(0, head)
        pre = root
        while head:
            node_val = head.val
            if node_val == val:
                pre.next = pre.next.next
            else:
                pre.next = head
                pre = pre.next
            head = head.next
        return root.next
```
