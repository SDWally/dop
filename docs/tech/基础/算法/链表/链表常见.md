# 链表常见

## 链表去重

```
class Solution:
    def deleteDuplicates(self, head: Optional[ListNode]) -> Optional[ListNode]:
        unique_set = set([])
        root = ListNode(0, head)
        pre = root
        while head:
            val = head.val
            # print(val) 
            if val not in unique_set:
                unique_set.add(val)
                pre.next = head
                pre = pre.next
            else:
                pre.next = pre.next.next
            head = head.next
        return root.next
```

## 去除指定元素

```
class Solution:
    def removeElements(self, head: Optional[ListNode], val: int) -> Optional[ListNode]:
        root = ListNode(0, head)
        pre = root
        while head:
            node_val = head.val
            if node_val == val:
                pre.next = pre.next.next
            else:
                pre.next = head
                pre = pre.next
            head = head.next
        return root.next
```

## 哈希集合

```

class MyHashSet:
    base = 769

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.ans = [[] for _ in range(MyHashSet.base)]

    def hash(self, key: int) -> int:
        return key % MyHashSet.base

    def add(self, key: int) -> None:
        h = self.hash(key)
        for element in self.ans[h]:
            if element == key:
                return
        self.ans[h].append(key)


    def remove(self, key: int) -> None:
        h = self.hash(key)
        for element in self.ans[h]:
            if element == key:
                self.ans[h].remove(key)
                return
        


    def contains(self, key: int) -> bool:
        """
        Returns true if this set contains the specified element
        """
        h = self.hash(key)
        for element in self.ans[h]:
            if element == key:
                return True
        return False

```

## 哈希映射

```

class MyHashMap:

    base = 769

    def __init__(self):
        self.ans = [[] for _ in range(MyHashMap.base)]

    def hash(self, key):
        return key % MyHashMap.base

    def put(self, key: int, value: int) -> None:
        h = self.hash(key)
        for i, (k, v) in enumerate(self.ans[h]):
            if k == key:
                self.ans[h][i] = (k, value)
                return
        self.ans[h].append((key, value))

    def get(self, key: int) -> int:
        h = self.hash(key)
        for k, v in self.ans[h]:
            if k == key:
                return v
        return -1

    def remove(self, key: int) -> None:
        h = self.hash(key)
        for k, v in self.ans[h]:
            if k == key:
                self.ans[h].remove((k,v))
                return
```

## 二进制链表转整数

```

class Solution:
    def getDecimalValue(self, head: ListNode) -> int:
        ans = 0
        while head:
            ans = ans * 2 + head.val
            head = head.next
        return ans
```

## 反向打印链表

```
class Solution:
    def reversePrint(self, head: ListNode) -> List[int]:
        val_list = []
        while head:
            val_list.append(head.val)
            head = head.next
        val_list.reverse()
        return val_list
```

## 链表中倒数第k个节点

```

class Solution:
    def getKthFromEnd(self, head: ListNode, k: int) -> ListNode:
        former = head
        latter = head
        for i in range(k):
            former = former.next
        while former:
            former = former.next 
            latter = latter.next
        return latter   
```
